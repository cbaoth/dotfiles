* [ ] Streamline the sourcing mechanism for commons.sh across scripts. Example:
. Keep one stable loader file (known path).
. Loader finds and sources commons.sh, validates required symbols:
+
[source,bash]
----
# ~/lib/commons-loader.sh: Locate and source commons.sh once
# code: language=bash insertSpaces=true tabSize=2
# -*- mode: sh; sh-shell: bash; indent-tabs-mode: nil; tab-width: 2 -*-
# vim: ft=bash:et:ts=2:sts=2:sw=2
# keywords: bash shell loader commons
# author: Andreas Weyer

cl::require_commons() {
  local -a candidates=(
    "${SCRIPT_PATH}/commons.sh"
    "${SCRIPT_PATH}/lib/commons.sh"
    "${HOME}/lib/commons.sh"
  )
  local f

  for f in "${candidates[@]}"; do
    if [[ -f "${f}" ]]; then
      # shellcheck source=/dev/null
      source "${f}"
      break
    fi
  done

  if ! command -v "cl::cmd_p" > /dev/null 2>&1; then
    printf "commons lib not found, exiting ..\n" >&2
    return 1
  fi

  return 0
}
----
. Scripts use a 1-liner source + 1-liner cl::require_commons call:
+
[source,bash]
----
# ...existing code...
source "${HOME}/lib/commons-loader.sh" || exit 1
cl::require_commons || exit 1
# ...existing code...
----
